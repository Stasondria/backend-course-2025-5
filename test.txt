# Тестування Частини 2

# 1. PUT - Завантажити тестову картинку у кеш
curl -X PUT http://127.0.0.1:3000/200 --data-binary @test_image.jpg

# 2. GET - Отримати картинку з кешу (повинно повернути картинку)
curl http://127.0.0.1:3000/200 --output test_200.jpg

# 3. GET - Спробувати отримати неіснуючу картинку (повинно повернути 404)
curl http://127.0.0.1:3000/404

# 4. DELETE - Видалити картинку з кешу
curl -X DELETE http://127.0.0.1:3000/200

# 5. GET - Спробувати отримати видалену картинку (повинно повернути 404)
curl http://127.0.0.1:3000/200

# 6. POST - Спробувати невідомий метод (повинно повернути 405)
curl -X POST http://127.0.0.1:3000/200

# Тестування Частини 3 - Проксі функціонал

# 7. GET - Завантажити картинку з http.cat (перший раз - з мережі)
curl http://127.0.0.1:3000/418 --output cat_418.jpg

# 8. GET - Завантажити ту ж картинку (другий раз - з кешу)
curl http://127.0.0.1:3000/418 --output cat_418_cached.jpg

# 9. GET - Спробувати неіснуючий HTTP код (повинно повернути 404)
curl http://127.0.0.1:3000/999

# 10. GET - Завантажити іншу картинку
curl http://127.0.0.1:3000/500 --output cat_500.jpg

# 11. Відкрити в браузері
# http://127.0.0.1:3000/200
# http://127.0.0.1:3000/404
# http://127.0.0.1:3000/418
